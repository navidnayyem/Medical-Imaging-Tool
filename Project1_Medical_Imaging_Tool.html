Step 1: Data Preparation¶

The labels of the X-ray images are given in the NLM-MontgomeryCXRSet-ReadMe.pdf file that contains in the dataset folder. data_dir = r'C:\Users\moham\OneDrive\Desktop\MontgomerySet\CXR_png'  # Path to the directory containing the images

# Initialize lists to hold image data and labels
data = []  
labels = []

# Loop through each image in the directory
for img_name in os.listdir(data_dir):
    img_path = os.path.join(data_dir, img_name)  # Get the full path of the image
    if img_name.endswith('_1.png'):
        label = 1  # Label for TB images
    elif img_name.endswith('_0.png'):
        label = 0  # Label for No-TB images
    else:
        continue  # Skip files that do not match the naming convention
    
    # Load the image and resize it to (100, 100)
    img = tf.keras.preprocessing.image.load_img(img_path, target_size=(100, 100))
    # Convert the image to an array
    img_array = tf.keras.preprocessing.image.img_to_array(img)
    data.append(img_array)  # Add the image array to the data list
    labels.append(label)  # Add the label to the labels list

# Convert lists to numpy arrays for easier manipulation
data = np.array(data)
labels = np.array(labels)

# Normalize the image data to the range [0, 1]
data = data / 255.0

def display_images(images, labels, num_images_per_label=3):
    plt.figure(figsize=(12, 6))
    
    # Get indices of images with label 0(No-TB) and 1(TB)
    indices_label_0 = [i for i, label in enumerate(labels) if label == 0]
    indices_label_1 = [i for i, label in enumerate(labels) if label == 1]
    
    # Randomly select indices
    selected_indices = random.sample(indices_label_0, num_images_per_label) + random.sample(indices_label_1, num_images_per_label)
    random.shuffle(selected_indices)
    
    for i, idx in enumerate(selected_indices):
        plt.subplot(1, num_images_per_label * 2, i + 1)
        plt.imshow(images[idx])
        plt.title(f"Label: {labels[idx]}")
        plt.axis('off')
    plt.show()

# Initialize a sequential model
model = Sequential([
    # Add a convolutional layer with 32 filters, a 3x3 kernel, ReLU activation, and input shape (100, 100, 3)
    Conv2D(32, (3, 3), activation='relu', input_shape=(100, 100, 3)),
    # Add a max pooling layer with 2x2 pool size
    MaxPooling2D((2, 2)),
    # Add another convolutional layer with 64 filters and ReLU activation
    Conv2D(64, (3, 3), activation='relu'),
    # Add another max pooling layer
    MaxPooling2D((2, 2)),
    # Add another convolutional layer with 128 filters and ReLU activation
    Conv2D(128, (3, 3), activation='relu'),
    # Add another max pooling layer
    MaxPooling2D((2, 2)),
    # Flatten the output from the convolutional layers
    Flatten(),
    # Add a dense (fully connected) layer with 128 units and ReLU activation
    Dense(128, activation='relu'),
    # Add a dropout layer to reduce overfitting (drop 50% of the units)
    Dropout(0.5),
    # Add a dense layer with 1 unit and sigmoid activation for binary classification
    Dense(1, activation='sigmoid')
])

# Compile the model with Adam optimizer, binary cross-entropy loss, and accuracy metric
model.compile(optimizer=Adam(learning_rate=0.001), loss='binary_crossentropy', metrics=['accuracy'])

# Function to display image with prediction
def display_image_with_prediction(image_path, model):
    img = cv2.imread(image_path)  # Read the image using OpenCV
    img = cv2.resize(img, (100, 100))  # Resize the image to match model input shape
    
    # Check if image is loaded successfully
    if img is None:
        print(f"Error: Unable to load image '{image_path}'.")
        return
    
    img = img.astype(np.float32) / 255.0  # Convert to float and normalize
    
    # Make prediction
    prediction = model.predict(np.expand_dims(img, axis=0))[0]
    if prediction >= 0.5:
        label = "TB Detected(1)"  # Label for TB detection
    else:
        label = "No-TB Detected(0)"  # Label for No-TB detection
    
    # Display image with prediction
    plt.imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))  # Convert BGR to RGB for displaying
    plt.title(f"Prediction: {label}")  # Set the title with the prediction
    plt.axis('off')  # Turn off axis
    plt.show()  # Show the image

# Function to randomly select and predict images from a folder
def predict_random_images(folder_path, model, num_images=5):
    image_files = os.listdir(folder_path)  # List all files in the folder
    random.shuffle(image_files)  # Shuffle the list of image files
    for filename in image_files[:num_images]:  # Loop through the specified number of images
        image_path = os.path.join(folder_path, filename)  # Get the full path of the image
        print(f"Filename: {filename}")  # Print the filename
        display_image_with_prediction(image_path, model)  # Display the image with prediction
